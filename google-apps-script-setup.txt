function doPost(e) {
  try {
    var sheet = SpreadsheetApp.openById("1uNnO_eZPXVYPKxvdzAjEFyhQIHeBagQBdhbMKUj_Skk").getSheetByName("Sheet1");
    
    // The form sends data via a hidden iframe with form-encoded POST.
    // The JSON payload is in the "payload" parameter.
    var data;
    if (e.parameter && e.parameter.payload) {
      // Hidden iframe form submission (form-encoded: key=payload, value=JSON)
      data = JSON.parse(e.parameter.payload);
    } else if (e.postData && e.postData.contents) {
      // Direct JSON POST (e.g., from fetch with Content-Type: application/json)
      data = JSON.parse(e.postData.contents);
    } else {
      throw new Error("No valid data received");
    }
    
    // Get the last row to calculate the next ID
    var lastRow = sheet.getLastRow();
    var nextId = 1;
    if (lastRow >= 5) {
      // Data starts at row 5 (row 4 is headers)
      var lastId = sheet.getRange(lastRow, 1).getValue();
      nextId = (parseInt(lastId) || 0) + 1;
    }
    
    // Format date and time separately (Philippine Time, UTC+8)
    var dateStr = Utilities.formatDate(new Date(), "Asia/Manila", "MM/dd/yyyy");
    var timeStr = Utilities.formatDate(new Date(), "Asia/Manila", "hh:mm:ss a");
    
    sheet.appendRow([
      nextId,
      dateStr,
      timeStr,
      data.firstName || "",
      data.lastName || "",
      data.email || "",
      data.phone || "",
      data.company || "",
      data.service || "",
      data.propertyType || "",
      data.message || ""
    ]);
    
    return ContentService
      .createTextOutput(JSON.stringify({ result: "success", id: nextId }))
      .setMimeType(ContentService.MimeType.JSON);
      
  } catch (error) {
    return ContentService
      .createTextOutput(JSON.stringify({ result: "error", error: error.toString() }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
